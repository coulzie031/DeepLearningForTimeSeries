\section{Dataset \& Problem}
% ─────────────────────────────────────────────────────────────────────────────

\begin{frame}{LSST Dataset — PLAsTiCC Challenge}
  \begin{columns}
    \begin{column}{0.55\textwidth}
      \textbf{Large Synoptic Survey Telescope}
      \begin{itemize}
        \item Classify \textbf{astronomical objects} from light curves
        \item Shape: $(N,\; T{=}36,\; C{=}6)$ — 6 photometric passbands
        \item \textbf{14 classes}: supernovae Ia/II/Ibc, AGN, variable stars,
              kilonovae, microlensing\ldots
        \item Train: \textbf{2,459} \quad Val: \textbf{492} \quad Test: \textbf{2,466}
      \end{itemize}
      \vspace{0.4em}
      \textbf{Two main difficulties}
      \begin{enumerate}
        \item \textbf{Missing values}: sparse telescope observations $\Rightarrow$ NaN
        \item \textbf{Severe imbalance}: class 53 has \textbf{only 7 train samples}\\
              imbalance ratio up to \textbf{155:1}
      \end{enumerate}
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{center}
      \footnotesize
      \begin{tabular}{lrr}
        \toprule
        Class (object type) & Train & \% \\
        \midrule
        90 — SN Ia (dominant) & 1,085 & 44.1 \\
        65 — variable star     &   431 & 17.5 \\
        42 — SN II             &   524 & 21.3 \\
        16 — TDE               &   370 & 15.1 \\
        \textcolor{myred}{53 — kilonovae} & \textcolor{myred}{7} & \textcolor{myred}{0.3} \\
        64 — AGN               &    24 &  1.0 \\
        \ldots                 & \ldots & \ldots \\
        \midrule
        \textbf{Total}         & \textbf{2,459} & 100 \\
        \bottomrule
      \end{tabular}
      \end{center}
      \vspace{0.4em}
      \small \textbf{SOTA} (Ruiz et al.\ 2021): MUSE = \textbf{63.6\%} accuracy
    \end{column}
  \end{columns}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}{LSST — Visual Structure of the Data}
  \begin{columns}
    \begin{column}{0.60\textwidth}
      \centering
      \safefig{figures/fig_lightcurves.png}{width=\linewidth}{8.5cm}
      {\footnotesize \textit{Sample light curves (8 classes). Each colour = one passband (u,g,r,i,z,y).}}
    \end{column}
    \begin{column}{0.38\textwidth}
      \textbf{What the model must learn:}
      \begin{itemize}
        \item \textbf{Shape}: supernovae (42, 90) rise fast then decay over $\sim$30 days
        \item \textbf{Periodicity}: variable stars (65) show oscillating flux
        \item \textbf{Duration}: TDE (16) can stay bright for months
        \item \textbf{Colour}: ratio across passbands (u–y) encodes temperature $\Rightarrow$ spectral type
      \end{itemize}
      \vspace{0.4em}
      \begin{block}{Challenge}
        \small Many classes look \emph{similar} in a single passband.
        The model must exploit all 6 passbands \emph{jointly} — this motivates our
        channel-concatenation strategy in Chronos.
      \end{block}
      \vspace{0.3em}
      \centering
      \safefig{figures/fig_dataset.png}{width=\linewidth}{4cm}
    \end{column}
  \end{columns}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}{Preprocessing \& Imbalance Handling}
  \begin{columns}
    \begin{column}{0.52\textwidth}
      \textbf{NaN handling}
      \begin{enumerate}
        \item Forward-fill $\to$ backward-fill $\to$ zero-fill
        \item Binary mask $(B, T, C)$: 1=observed, 0=missing
      \end{enumerate}
      \vspace{0.4em}
      \textbf{Normalisation}
      \begin{itemize}
        \item Z-score per channel, fit on train only
      \end{itemize}
      \vspace{0.4em}
      \textbf{Imbalance: three-level strategy}
      \begin{enumerate}
        \item \texttt{WeightedRandomSampler} (inverse-freq)
        \item \textbf{Focal Loss} ($\gamma=2$): down-weights easy majority examples
        \item F1-weighted ensemble voting
      \end{enumerate}
    \end{column}
    \begin{column}{0.45\textwidth}
      \textbf{Augmentation (train only)}
      \begin{itemize}
        \item Jitter ($\sigma=0.03$), scaling ($\times[0.9,1.1]$)
        \item Channel dropout ($p=0.15$)
        \item MixUp ($\alpha=0.3$)
      \end{itemize}
      \vspace{0.5em}
      \begin{block}{Key insight}
        \small Most bake-off methods (MUSE, ROCKET)
        do \emph{not} handle class imbalance.
        Our WeightedSampler alone adds
        \textbf{$\sim$8\%} accuracy over naive training.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\section{Baseline: InceptionTime}
% ─────────────────────────────────────────────────────────────────────────────

\begin{frame}{Baseline — InceptionTime (from scratch)}
  \begin{columns}
    \begin{column}{0.50\textwidth}
      \textbf{Architecture}
      \begin{itemize}
        \item 3 Inception blocks, $F=32$ filters, multi-scale kernels: 9, 19, 39
        \item Residual connections + Global Avg Pooling
        \item \textbf{474k parameters}
      \end{itemize}
      \vspace{0.4em}
      \textbf{Training}
      \begin{itemize}
        \item Focal Loss ($\gamma=2$) + label smoothing $\epsilon=0.1$
        \item AdamW + CosineAnnealing, early stop on macro F1
        \item Best val F1 at epoch \textbf{29}
      \end{itemize}
      \vspace{0.4em}
      \begin{alertblock}{Result}
        \centering Acc = \textbf{0.5483} \quad Macro F1 = \textbf{0.3753}
      \end{alertblock}
    \end{column}
    \begin{column}{0.48\textwidth}
      \centering
      \safefig{figures/fig_curves.png}{width=\linewidth}{7cm}
      \vspace{0.2em}
      {\footnotesize \textit{Training curves — visible overfitting on accuracy;\\

\end{frame}

\section{Foundation Models (Setting 1)}
% ─────────────────────────────────────────────────────────────────────────────

\begin{frame}{Setting 1 — Adapting Foundation Models}
  \begin{block}{What is a time-series foundation model?}
    A large pre-trained model (on billions of timesteps from diverse domains)
    adapted to a new downstream task via fine-tuning.
  \end{block}
  \vspace{0.5em}
  \textbf{We tested two foundation models:}
  \vspace{0.3em}
  \begin{center}
  \begin{tabular}{lcccc}
    \toprule
    Model & Params & Pre-training & Classification? & Our result \\
    \midrule
    \textbf{MOMENT-1-large} & 341M & 1B+ TS (diverse) & Native & F1=0.279 \\
    \textcolor{myblue}{\textbf{Chronos-T5-Small}} & \textcolor{myblue}{46M} & \textcolor{myblue}{Real + synthetic} & \textcolor{myblue}{Adapted} & \textcolor{myblue}{\textbf{F1=0.433}} \\
    \bottomrule
  \end{tabular}
  \end{center}
  \vspace{0.5em}
  \begin{columns}
    \begin{column}{0.48\textwidth}
      \begin{block}{MOMENT (ICML 2024)}
        \small T5-based encoder, natively supports classification.
        Channel-independent: each passband $\to$ 1024-dim embedding.
        Stack 6 channels $\to$ 6144 features.
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Chronos (NeurIPS 2024, Amazon)}
        \small Forecasting model — but its T5 encoder captures
        temporal patterns usable for classification.
        ``Transfer from forecasting to classification.''
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}{Approach 1 — MOMENT-1-large (341M params)}
  \begin{columns}
    \begin{column}{0.55\textwidth}
      \textbf{Adaptation}
      \begin{itemize}
        \item Pad $T=36 \to 40$ (multiple of patch length 8)
        \item Channel-flatten: $(B,6,1024) \to (B,6144)$
        \item MLP classification head on top
      \end{itemize}
      \vspace{0.4em}
      \textbf{Phase 1 — Linear Probing (150 ep)}
      \begin{itemize}
        \item Backbone frozen, only head trains ($\sim$594k params)
        \item Best val F1 = 0.3030 at ep62
      \end{itemize}
      \vspace{0.3em}
      \textbf{Phase 2 — Partial Unfreeze (26 ep)}
      \begin{itemize}
        \item Last 4/24 T5 encoder blocks unfrozen (51M params)
        \item LR$_\text{enc}=5\times10^{-6}$, linear warmup
        \item Best val F1 = 0.3030 (no improvement)
      \end{itemize}
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{alertblock}{MOMENT results (test set)}
        \begin{center}
        \begin{tabular}{ll}
          \toprule
          Accuracy & 0.3078 \\
          Macro F1 & \textbf{0.2791} \\
          \bottomrule
        \end{tabular}
        \end{center}
      \end{alertblock}
      \vspace{0.4em}
      \textbf{Why does MOMENT underperform?}
      \begin{itemize}
        \item 341M params, only 1,967 train samples $\Rightarrow$ difficult regime
        \item T5 architecture very sensitive to LR in fine-tuning
        \item Phase 2 fails to improve: early stop at ep26
        \item \textbf{Excluded from ensemble} (val F1=0.303 $<$ threshold 0.33)
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}{Approach 2 — Chronos-T5-Small (46M params)}
  \begin{columns}
    \begin{column}{0.55\textwidth}
      \textbf{Chronos: originally a forecasting model}
      \begin{itemize}
        \item Pre-trained to predict future values via tokenised quantiles
        \item We repurpose only the \textbf{T5 encoder} (discard decoder)
        \item Analogous to using BERT for sentence classification
      \end{itemize}
      \vspace{0.4em}
      \textbf{Adaptation to LSST}
      \begin{itemize}
        \item Each passband processed \textbf{independently} through Chronos tokeniser
        \item Masked mean-pooling over encoder hidden states $\to$ 512-dim per channel
        \item Concatenate 6 channels: $6 \times 512 = \mathbf{3072}$-dim
        \item 3-layer MLP classification head
      \end{itemize}
      \vspace{0.4em}
      \textbf{Phase 1 — Linear Probing (50 ep):} val F1=0.4292\\
      \textbf{Phase 2 — Unfreeze last 4 blocks (40 ep):} val F1=\textbf{0.4333}
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{block}{Chronos results (test set)}
        \begin{center}
        \begin{tabular}{ll}
          \toprule
          Phase 1 accuracy & 0.5211 \\
          Phase 1 Macro F1 & 0.4292 \\
          \midrule
          \textbf{Phase 2 accuracy} & \textbf{0.5333} \\
          \textbf{Phase 2 Macro F1} & \textbf{0.4333} \\
          \bottomrule
        \end{tabular}
        \end{center}
      \end{block}
      \vspace{0.4em}
      \textbf{Why does Chronos succeed?}
      \begin{itemize}
        \item \textbf{46M} params: well-sized for 1,967 train samples
        \item Chronos tokeniser handles NaN naturally
        \item Forecasting pre-training captures temporal dynamics
        \item F1=\textbf{0.43} $>$ InceptionTime baseline (0.38)
      \end{itemize}
      \vspace{0.2em}
      \small \textcolor{mygreen}{\textbf{$\checkmark$ Included in ensemble}}
    \end{column}
  \end{columns}
\end{frame}

% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}{Chronos Tokenisation Pipeline — How it Works}
  \begin{columns}
    \begin{column}{0.52\textwidth}
      \textbf{Step-by-step adaptation}
      \vspace{0.3em}

      \begin{enumerate}
        \setlength\itemsep{0.35em}
        \item \textbf{Input}: $(B, T{=}36, C{=}6)$ multivariate series

        \item \textbf{Split} into 6 univariate channels:\\
              $\{x_c\}_{c=1}^{6}$, each of shape $(B, 36)$

        \item \textbf{Chronos tokeniser} per channel:\\
              maps continuous values $\to$ quantile bin IDs
              (4096-token vocabulary)

        \item \textbf{T5 encoder} $\to$ hidden states $(B, L, 512)$

        \item \textbf{Masked mean-pool} over observed positions:\\
              $\mathbf{e}_c = \frac{\sum_t m_t \, h_t}{\sum_t m_t} \in \mathbb{R}^{512}$

        \item \textbf{Concatenate} all channels:\\
              $\mathbf{z} = [\mathbf{e}_1 \| \ldots \| \mathbf{e}_6] \in \mathbb{R}^{3072}$

        \item \textbf{MLP head}: $\mathbb{R}^{3072} \to \mathbb{R}^{14}$ (Softmax)
      \end{enumerate}
    \end{column}
    \begin{column}{0.46\textwidth}
      \vspace{0.2em}
      \begin{center}
      \footnotesize
      \renewcommand{\arraystretch}{1.3}
      \begin{tabular}{c}
        \rowcolor{blue!10}
        \fbox{\parbox{4.2cm}{\centering Input $(B,36,6)$}} \\
        $\downarrow$ \\
        \rowcolor{orange!15}
        \fbox{\parbox{4.2cm}{\centering Split $\to$ 6 channels $(B,36)$ each}} \\
        $\downarrow$ \\
        \rowcolor{orange!25}
        \fbox{\parbox{4.2cm}{\centering Chronos tokeniser (quantile bins)}} \\
        $\downarrow$ \\
        \rowcolor{blue!15}
        \fbox{\parbox{4.2cm}{\centering T5 Encoder\\{\tiny frozen Ph.1 / partial Ph.2}}} \\
        $\downarrow$ \\
        \rowcolor{blue!8}
        \fbox{\parbox{4.2cm}{\centering Masked mean-pool $\to$ 512-dim/ch}} \\
        $\downarrow$ \\
        \rowcolor{green!15}
        \fbox{\parbox{4.2cm}{\centering Concat $6\times512=3072$-dim}} \\
        $\downarrow$ \\
        \rowcolor{green!25}
        \fbox{\parbox{4.2cm}{\centering MLP head $\to$ 14 classes}} \\
      \end{tabular}
      \end{center}
      \vspace{0.2em}
      \begin{block}{Why channel-independent?}
        \small Chronos is \emph{univariate} by design. Processing each passband
        separately preserves the pre-trained tokeniser semantics.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

  \begin{center}
  \begin{tabular}{lcccc}
    \toprule
    & \textbf{MOMENT} & \textbf{Chronos} & Advantage \\
    \midrule
    Parameters     & 341M & \textbf{46M} & Chronos (right-sized) \\
    Pre-training   & Classification (native) & Forecasting (adapted) & MOMENT \\
    Phase 1 F1     & 0.303 & \textbf{0.429} & \textbf{Chronos +12.6pts} \\
    Phase 2 F1     & 0.279 & \textbf{0.433} & \textbf{Chronos +15.4pts} \\

    Training time  & $\sim$30 min & \textbf{$\sim$15 min} & Chronos \\
    \bottomrule
  \end{tabular}
  \end{center}
  \vspace{0.5em}
  \begin{block}{Key lesson}
    A \textbf{smaller, well-matched} foundation model (Chronos, 46M)
    outperforms a \textbf{larger but oversized} one (MOMENT, 341M)
    when training data is scarce (1,967 samples).
    Domain gap matters less than model scale.
  \end{block}
\end{frame}

\section{Ensemble Pipeline}
% ─────────────────────────────────────────────────────────────────────────────

\begin{frame}{Ensemble Pipeline — 4 Diverse Models}
  \begin{columns}
    \begin{column}{0.52\textwidth}
      \textbf{Members:}
      \begin{enumerate}
        \item \textcolor{myblue}{\textbf{Chronos-T5-Small}} (foundation model, 2-phase)
        \item \textbf{InceptionTime-Large $\times 5$} ($F=64$, 5 seeds)
        \item \textbf{PatchTST} ($d=64$, 4 heads, 3 layers)
        \item \textbf{MultiROCKET} (79,968 random kernel features)
      \end{enumerate}
      \vspace{0.4em}
      \textbf{F1-weighted soft voting:}
      $$w_i = \frac{\exp(10 \cdot F1_{\text{val},i})}{\sum_j \exp(10 \cdot F1_{\text{val},j})}$$
      Members with val F1 $<$ 0.33 excluded.\\[0.4em]
      \textbf{TTA:} average $N=5$ augmented test passes
      for IT$\times$5 and PatchTST (+1--2\% free).
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{center}
      \footnotesize
      \begin{tabular}{lccc}
        \toprule
        Model & Val F1 & Weight & In? \\
        \midrule
        IT-Large$\times$5  & 0.7154 & \textbf{0.852} & \textcolor{mygreen}{Yes} \\
        PatchTST           & 0.4626 & 0.068 & \textcolor{mygreen}{Yes} \\
        \textcolor{myblue}{Chronos} & \textcolor{myblue}{0.4333} & 0.051 & \textcolor{mygreen}{Yes} \\
        MultiROCKET        & 0.3782 & 0.029 & \textcolor{mygreen}{Yes} \\
        MOMENT             & 0.3030 & — & \textcolor{myred}{No} \\
        \bottomrule
      \end{tabular}
      \end{center}
      \vspace{0.4em}
      \begin{block}{Why IT$\times$5 dominates?}
        \small Training 5 seeds and averaging is the standard
        InceptionTime recommendation (Fawaz et al.\ 2020).
        It achieves the highest val F1 on this dataset.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

        \item \textbf{Chronos Phase 2} unlocks +1.5pts F1 over linear probing
        \item Each ensemble member adds \textbf{marginal but consistent} gains
        \item \textbf{Diversity matters}: kernel method (MultiROCKET) contributes
              even with low val F1 (0.38), because it captures different features
              than CNN/Transformer members
      \end{itemize}
      \vspace{0.3em}
      \begin{alertblock}{MUSE comparison}
        \small MUSE uses no imbalance handling. On a balanced dataset it would
        likely score lower than our ensemble.
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\section{Conclusion}
% ─────────────────────────────────────────────────────────────────────────────

\begin{frame}{Conclusion \& Contributions}
  \begin{columns}
    \begin{column}{0.55\textwidth}
      \textbf{Summary}
      \begin{enumerate}
        \item \textbf{Preprocessing matters}: WeightedSampler + Focal Loss critical for imbalanced LSST
        \item \textbf{First application} of Chronos-T5-Small to multivariate TS classification
        \item \textbf{Chronos F1=0.433} outperforms MOMENT (0.279) and baseline (0.375)
          despite Chronos being a \emph{forecasting} model — demonstrates transfer learning versatility
        \item \textbf{Ensemble Acc=0.6448} beats MUSE SOTA (0.636):
              $+$0.86\% with no extra data
      \end{enumerate}
      \vspace{0.4em}
      \begin{block}{Replication}
        Full code available. Single Colab T4 GPU, $\sim$2h total runtime.
      \end{block}
    \end{column}
    \begin{column}{0.43\textwidth}
      \textbf{Limitations}
      \begin{itemize}
        \item Classes 53 (7 samples), 64 (24 samples): F1=0 — fundamental data limitation
        \item MOMENT Phase 2 fails to improve: T5 unstable with few data
        \item Ensemble dominated by IT$\times$5 (weight=0.85) — Chronos contributes marginally to final accuracy
      \end{itemize}
      \vspace{0.4em}
      \textbf{Future work}
      \begin{itemize}
        \item Chronos-T5-Large (200M) or Tiny (8M) — size ablation
        \item Few-shot meta-learning for classes 53, 64
        \item Apply to real LSST survey data (not simulated PLAsTiCC)
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
